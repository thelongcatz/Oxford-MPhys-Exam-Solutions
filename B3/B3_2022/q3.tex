	\part From TDSE:
	\begin{gather*}
		i\hbar \pdiff{t}\Psi = \hat{H}\Psi \\
		\Rightarrow \sbracket{\splitfrac{\dot{c}_1 \psi_1 e^{-i\omega_1 t} + c_1 \psi_1 (-i\omega_1) e^{-i\omega_1 t}}{+\dot{c}_2 \psi_2 e^{-i\omega_2 t} + c_2 \psi_2 (-i\omega_2) e^{-i\omega_2 t}}} = \splitfrac{\rbracket{\hat{H}_0 + \hat{D}_0 \cos (\omega t)}}{\sbracket{c_1 \psi_1 e^{-i\omega_1 t} + c_2 \psi_2 e^{-i\omega_2 t}}} \mtext{where } \omega_i = \frac{E_i}{\hbar} \\
		\splitfrac{i\hbar \dot{c}_1 \psi_1 e^{-i\omega_1 t} + \cancel{\hbar\omega_1 c_1 \psi_1 e^{-i\omega_1 t}}}{i\hbar \dot{c}_2 \psi_2 e^{-i\omega_2 t} + \cancel{\hbar\omega_2 c_2 \psi_2 e^{-i\omega_2 t}}} = \splitfrac{\cancel{E_1 c_1 \psi_1 e^{-i\omega_1 t}} + \cancel{E_2 c_2 \psi_2 e^{-i\omega_2 t}}}{+\hat{D}_0 \cos (\omega t) \sbracket{c_1 \psi_1 e^{-i\omega_1 t} + c_2 \psi_2 e^{-i\omega_2 t}}} \\
		\xRightarrow{\times\psi_1^*} i\hbar\dot{c}_1 \underbracket{|\psi_1|^2}_{\mathclap{\substack{\textnormal{1 assuming} \\ \textnormal{normalised}}}} e^{-i\omega_1 t} = c_1 e^{-i\omega_1 t} \cancelto{0}{\bra{\psi_1}\hat{D}_0 \cos (\omega t)\ket{\psi_1}} + c_2 e^{-i\omega_2 t} \bra{\psi_1}\hat{D}_0 \cos (\omega t)\ket{\psi_2} \\
		\Rightarrow \dot{c}_1 = -\frac{i}{\hbar} c_2 e^{-i\omega_0 t} V_{12} \cos (\omega t) \mtext{where } V_{ij} = \bra{\psi_i} \hat{D}_0 \ket{\psi_j}
	\end{gather*}
	
	Similarly:
	\begin{align*}
		\dot{c}_2 &= -\frac{i}{\hbar} c_1 e^{i\omega_0 t} V_{21} \cos (\omega t) \\
		&= -\frac{i}{2\hbar} c_1 V_{21} \rbracket{e^{i(\omega_0 + \omega)t} + e^{i(\omega_0 - \omega)t}}
	\end{align*}
	
	For $|\omega - \omega_0| \ll \omega_0$, we may invoke rotating wave approximation and drop the term with $\omega + \omega_0$:
	\begin{align*}
		\dot{c}_1 &= -\frac{i}{2\hbar} c_2 V_{12} e^{-i(\omega_0 - \omega)t} \\
		\dot{c}_2 &= -\frac{i}{2\hbar} c_1 V_{21} e^{i(\omega_0 - \omega)t} \\
		\Rightarrow \ddot{c}_2 &= -\frac{i}{2\hbar} \dot{c}_1 V_{21} e^{i(\omega_0 - \omega)t} + i(\omega_0 - \omega)\dot{c}_2 \\
		0 &= \ddot{c}_2 - i(\omega - \omega_0)\dot{c}_2 + \rbracket{\frac{1}{2\hbar}}^2 c_2 \underbracket{V_{12} V_{21}}_{|V|^2} e^{0}
	\end{align*}
	
	Try ansatz $c_2 = A e^{i\alpha t}$:
	\begin{align*}
		\Rightarrow -\alpha^2 + \alpha (\omega - \omega_0) + \rbracket{\frac{1}{2\hbar}}^2 |V|^2 &= 0 \\
		\alpha &= \frac{(\omega - \omega_0) \pm \sqrt{(\omega - \omega_0)^2 + 4 \rbracket{\frac{V}{2\hbar}}^2}}{2} \\
		&= \frac{\delta \pm \sqrt{\delta^2 + \Omega^2}}{2} \mtext{where $\delta = \omega - \omega_0$, $\Omega = \dfrac{V}{\hbar}$} \\
		&= \frac{\delta + \sqrt{\delta^2 + \Omega^2}}{2} \mtext{since frequency $>0$}
	\end{align*}
	
	So $c_2 = A e^{i\alpha t} + B e^{-i\alpha t}$, $c_2(0)=0$:
	\begin{align*}
		\Rightarrow A &= -B \\
		\Rightarrow c_2(t) &= A \sin (\alpha t) \\
		\Rightarrow \dot{c}_2(t) &= A\alpha \cos (\alpha t) \\
		&= -\frac{i}{2\hbar} c_1 V e^{i\delta t} \\[1em]
		c_1(t) &= \frac{2i\hbar A\alpha}{V} \cos (\alpha t) e^{-i\delta t} \\
		c_1(0) &= 1 \\
		\Rightarrow A &= \frac{V}{2i\hbar\alpha} = \frac{\Omega}{2\alpha i}
	\end{align*}
	
	So:
	\begin{align*}
		|c_2(t)|^2 &= \rbracket{\frac{\Omega}{2\alpha}}^2 \sin^2 (\alpha t) \\
		&= \rbracket{\frac{\Omega}{\delta + \sqrt{\delta^2 + \Omega^2}}}^2 \sin^2 \rbracket{\frac{1}{2} (\delta + \sqrt{\delta^2 + \Omega^2}) t}
	\end{align*}
	
	Time average:
	\begin{align*}
		\bar{c}_2 &= \frac{1}{\tau} \defint{0}{\tau}{\rbracket{\frac{\Omega}{2\alpha}}^2 \sin^2 \rbracket{\frac{1}{2}\alpha t}}{t} \\
		&= \frac{1}{\tau} \rbracket{\frac{\Omega}{2\alpha}}^2 \defint{0}{\tau}{\frac{1-\cos (\alpha t)}{2}}{t} \\
		&= \frac{1}{2\tau} \rbracket{\frac{\Omega}{2\alpha}}^2 \sbracket{t - \frac{1}{\alpha} \sin (\alpha t)}_{t=0}^\tau \\
		&= \frac{1}{2} \rbracket{\frac{\Omega}{2\alpha}}^2 \sbracket{1-\frac{\sin (\alpha\tau)}{\alpha\tau}}
	\end{align*}
	As $\tau\to\infty$, $\bar{c}_2 \to \dfrac{1}{2} \rbracket{\dfrac{\Omega}{2\alpha}}^2$.

	\part \todo Decay term $e^{-\frac{\gamma t}{2}}$, where $\gamma = \dfrac{1}{\tau}$ is decay rate, should be added to $c_2(t)$ to account for natural broadening.
	
	Fourier transforming the decay term should give broadening in frequency domain:
	\begin{align*}
		\defint{0}{\infty}{\frac{1}{\pi} e^{i\omega t} e^{-\frac{\gamma t}{2}}}{t} &= \frac{1}{\pi} \sbracket{\frac{e^{(-\frac{\gamma}{2}+i\omega)t}}{i\omega - \cfrac{\gamma}{2}}}_{t=0}^\infty \\
		&= \frac{1}{\pi} \frac{1}{\cfrac{\gamma}{2} - i\omega} \\
		&= \frac{1}{\pi} \frac{\cfrac{\gamma}{2} + i\omega}{\rbracket{\cfrac{\gamma}{2}}^2 + \omega^2} \\
		&\rightarrow \frac{1}{\pi} \frac{\cfrac{\gamma}{2}}{\rbracket{\cfrac{\gamma}{2}}^2 + \omega^2} \mtext{by taking Re() component}
	\end{align*}
	
	Hence natural broadening forms a Lorentzian in frequency domain.
	
	Another example of Lorentzian broadening is phonon broadening.
	
	\part With the addition of $e^{-b|t|}$, the differentials now read:
	\begin{align*}
		\dot{c}_1 &= -\frac{i}{2\hbar} c_2 V e^{-i(\omega_0 - \omega)t} e^{-b|t|} \\
		\dot{c}_2 &= -\frac{i}{2\hbar} c_1 V e^{i(\omega_0 - \omega)t} e^{-b|t|}
	\end{align*}
	
	Assuming weak radiation, then $c_1 \simeq 1$ and $c_2 \simeq 0$.
	\begin{align*}
		\dot{c}_2 &= -\frac{i}{2\hbar} V e^{i(\omega_0 - \omega)t} e^{-b|t|} \\
		\Rightarrow c_2(t) &= \defint{-\infty}{t}{-\frac{i}{2\hbar} V e^{-b|t^\prime|+i(\omega_0 - \omega)t^\prime}}{t^\prime} \\
		&= -\frac{i}{2\hbar} V \frac{e^{(-b+i(\omega_0 - \omega))t}}{-b+i(\omega_0 - \omega)} \\
		&= \underbracket{-\frac{i}{2} \Omega \frac{b+i(\omega_0 - \omega)}{b^2 + (\omega_0 - \omega)^2}}_{\mathclap{-\frac{1}{2} \Omega \frac{ib - (\omega_0 - \omega)}{b^2 + (\omega_0 - \omega)^2}}} e^{-bt} e^{i(\omega_0 - \omega)t}
	\end{align*}
	Hence $|c_2(t)|^2 = \dfrac{1}{4} \Omega^2 \dfrac{1}{b^2 + \delta^2} e^{-2bt}$.
	
	As $t\to\infty$, $|c_2(t)|^2 \to \dfrac{1}{4} \Omega^2 \dfrac{1}{b^2 + \delta^2}$.
	
	Hence the excitation probability is a Lorentzian with FWHM $2b$.
