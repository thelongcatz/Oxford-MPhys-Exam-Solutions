	\part
	\begin{align*}
		\hat{C}\hat{P}\ket{+} &= \ket{+} = \frac{1}{\sqrt{2}} \rbracket{\ket{\Kz} + \ket{\bar{\Kz}}} \\
		\hat{C}\hat{P}\ket{-} &= -\ket{-} = -\frac{1}{\sqrt{2}} \rbracket{\ket{\Kz} - \ket{\bar{\Kz}}}
	\end{align*}
	
	\begin{align*}
		\hat{P}\ket{+} = \hat{P} \sbracket{\frac{1}{\sqrt{2}}\rbracket{\ket{\Kz} + \ket{\bar{\Kz}}}} \\
		&= -\frac{1}{\sqrt{2}} \rbracket{\ket{\Kz} + \ket{\bar{\Kz}}}
	\end{align*}
	since kaons are fermions.
	
	Similarly,
	\begin{equation*}
		\hat{P}\ket{-} = -\frac{1}{\sqrt{2}} \rbracket{\ket{\Kz} - \ket{\bar{\Kz}}}
	\end{equation*}
	
	So $\hat{C}$ should introduce a negative parity to the overall wavefunction while exchanging $\Kz$ with $\bar{\Kz}$:
	\begin{align*}
		\hat{C}\hat{P}\ket{+} &= \hat{C}\sbracket{-\frac{1}{\sqrt{2}} \rbracket{\ket{\Kz} + \ket{\bar{\Kz}}}} \\
		&= \frac{1}{\sqrt{2}} \rbracket{\ket{\bar{\Kz}} + \ket{\Kz}} = \ket{+} \\[1em]
		\hat{C}\hat{P}\ket{-} &= \hat{C}\sbracket{-\frac{1}{\sqrt{2}} \rbracket{\ket{\bar{\Kz}} - \ket{\Kz}}} \\
		&= \frac{1}{\sqrt{2}} \rbracket{\ket{\bar{\Kz}} - \ket{\Kz}} = -\ket{-}
	\end{align*}
	CP symmetry is obeyed under strong interaction, so the final state must have the same CP parity.
	
	Each pion carries intrinsic parity $-1$, so $\ket{+}$ should decay into 2 pions, $\ket{-}$ into 3 pions (1 pion would be kinematically unfavourable due to the mass difference).
	{\color{red}Also no way to conserve momentum in CoM frame.}
	
	\begin{equation*}
		\hat{C}\hat{P}\ket{\pi} = -\ket{\pi}
	\end{equation*}
	So $\ket{+} \to$ even $\pi$'s, $\ket{-} \to$ odd $\pi$'s.

	\begin{equation*}
		\ket{\Kz} = \frac{1}{\sqrt{2}} \rbracket{\ket{+} + \ket{-}} \qquad \ket{\bar{\Kz}} = \frac{1}{\sqrt{2}} \rbracket{\ket{+} - \ket{-}}
	\end{equation*}
	
	\begin{align*}
		\bra{\mathbf{r}}\ket{\Kz} &= \frac{1}{\sqrt{2}} \rbracket{\bra{\mathbf{r}}\ket{+} + \bra{\mathbf{r}}\ket{-}} \\
		&= \frac{1}{\sqrt{2}} \sbracket{A_S(t) + A_L(t)} \\[1em]
		\bra{\mathbf{r}}\ket{\bar{\Kz}} &= \frac{1}{\sqrt{2}} \sbracket{A_S(t) - A_L(t)} \mtext{similarly}
	\end{align*}
	
	\part
	\begin{align*}
		I &= \abs{\bra{\mathbf{r}}\ket{\Kz}}^2 \\
		&= \frac{1}{2} \sbracket{|A_S(t)|^2 + A_S^*(t) A_L(t) + A_S(t) A_L^*(t) + |A_L(t)|^2} \\
		&= \frac{1}{2} \sbracket{\splitfrac{\rbracket{A_S(0)}^2 e^{-\frac{t}{\tau_s}} + A_S^*(0) A_L(0) e^{i(m_S - m_L)t} e^{-\frac{t}{2\tau_s}} e^{-\frac{t}{2\tau_L}}}{+ A_S(0) A_L^*(0) e^{-i(m_S - m_L)t} e^{-\frac{t}{2\tau_s}} e^{-\frac{t}{2\tau_L}} + \rbracket{A_L(0)}^2 e^{-\frac{t}{\tau_L}}}} \\
		&= \underbracket{\frac{A_S(0)^2}{2}}_{A} e^{-\frac{t}{\tau_s}} + \underbracket{A_S(0) A_L(0) e^{-\frac{t}{2\tau_s} e^{-\frac{t}{2\tau_L}}}}_{C} \cos\underbracket{(m_S - m_L)t}_{\phi} + \underbracket{\frac{A_L(0)^2}{2}}_{B} e^{-\frac{t}{\tau_L}}
	\end{align*}

	\end{align*}
