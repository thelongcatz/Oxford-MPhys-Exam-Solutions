	\part Total momentum of the system:
	\begin{equation*}
		\mathsf{P}^\mu = \rbracket{\sum_i E_i, \sum_i \mathbf{p}_i}
	\end{equation*}
	
	Consider a transformation to the ZMF (align $x$-axis with $\sum_i \mathbf{p}_i$):
	\begin{align*}
		\rbracket{\mathsf{P}^\prime}^\mu &= \mathsf{\Lambda}^\mu_\nu \mathsf{P}^\nu \\
		\Rightarrow \rbracket{\mathsf{P}^\prime}^1 &= 0 = -\beta\gamma\rbracket{\sum_i E_i} + \gamma\rbracket{\sum_i \mathbf{p}_i} \\
		\Rightarrow \beta &= \frac{\sum_i \mathbf{p}_i}{\sum_i E_i}
	\end{align*}
	
	\part Sketch of the decay process:
	\image{.8\linewidth}{q2-decay}
	Conservation of 4-momentum gives:
	\begin{align*}
		\mathsf{P}_\pi^\mu &= \rbracket{\mathsf{P}_1^\mu + \mathsf{P}_2^\mu} \\
		\Rightarrow \rbracket{\mathsf{P}_\pi^\mu}^2 &= \rbracket{\mathsf{P}_1^\mu + \mathsf{P}_2^\mu}^2 \\
		-M_\pi^2 &= \cancelto{0}{\rbracket{\mathsf{P}_1^\mu}^2} + \cancelto{0}{\rbracket{\mathsf{P}_2^\mu}^2} + 2\rbracket{\mathsf{P}_1}_\mu \rbracket{\mathsf{P}_2}^\mu \\
		&= 2\rbracket{-E_1 E_2 + p_1 p_2 \cos A} = 2p_1 p_2 \rbracket{\cos A - 1} \mtext{since $E=pc$ for $\gamma$}
	\end{align*}
	
	With $\cos 2\theta = \cos^2 \theta - \sin^2 \theta = 1-2\sin^2 \theta$, we then have:
	\begin{align*}
		\Rightarrow M_\pi^2 &= -2E_1 E_2 \rbracket{\cancel{1} - 2\sin^2 \frac{A}{2} \cancel{-1}} \\
		\Rightarrow \sin^2 \frac{A}{2} &= \frac{M_\pi^2}{4E_1 E_2}
	\end{align*}

		&= a \frac{\sin\theta^* \inftsml{\theta^*}}{\inftsml{A}} = a\deri{\rbracket{\cos\theta^*}}{A}
	\end{align*}
